<script>
function esc(s){if(s==null)return"";return String(s)
    .replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;").replace(/"/g,"&quot;");}

var DATA;
try{
    var raw=document.getElementById("test-data").textContent.trim();
    DATA=(raw.indexOf("{{")===0||raw==="")?null:JSON.parse(raw);
}catch(e){DATA=null;}

function selectedIds(){
    var ids=[];
    var inputs=document.querySelectorAll(".opt input");
    for(var i=0;i<inputs.length;i++){if(inputs[i].checked)ids.push(inputs[i].getAttribute("data-id"));}
    return ids;
}

function refresh(){
    var ids=selectedIds();
    var btn=document.getElementById("btn-answer");
    if(btn){
        btn.setAttribute("href","cmd:answer:"+ids.join(","));
        if(ids.length===0)btn.classList.add("disabled"); else btn.classList.remove("disabled");
    }
    var rows=document.querySelectorAll(".opt");
    for(var i=0;i<rows.length;i++){
        var inp=rows[i].querySelector("input");
        if(inp&&inp.checked)rows[i].classList.add("sel"); else rows[i].classList.remove("sel");
    }
}

function render(){
    var root=document.getElementById("root");
    if(!DATA){root.innerHTML='<div class="card info">'+esc("")+'</div>';return;}
    var L=DATA.labels||{};
    var html="";

    html+='<div class="topbar">'
        +'<div class="topbar-title">'+esc(L.title)+'</div>'
        +'<div class="metrics">'
        +'<div class="metric">'+esc(L.task)+'<b>'+esc(DATA.num)
        +(DATA.limit>0?(" "+esc(L.of)+" "+esc(DATA.limit)):"")+'</b></div>'
        +'<div class="metric">'+esc(L.theta)+'<b>'+esc(DATA.theta)+'</b></div>'
        +'<div class="metric">'+esc(L.se)+'<b>'+esc(DATA.se)+'</b></div>'
        +'</div></div>';

    html+='<div class="card">'
        +'<div class="q-label">'+esc(L.question)+'</div>'
        +'<div class="q-text">'+esc(DATA.question)+'</div>';

    if(DATA.active){
        var opts=DATA.options||[];
        if(opts.length>0){
            html+='<div class="opts-hint">'+esc(DATA.multiple?L.pickMany:L.pickOne)+'</div>';
            html+='<div class="opts">';
            var type=DATA.multiple?"checkbox":"radio";
            for(var i=0;i<opts.length;i++){
                html+='<label class="opt">'
                    +'<input type="'+type+'" name="opt" data-id="'+esc(opts[i].id)+'" onchange="refresh()">'
                    +'<span class="otext">'+esc(opts[i].text)+'</span></label>';
            }
            html+='</div>';
            html+='<div class="actions">'
                +'<a id="btn-answer" class="btn btn-primary disabled" href="cmd:answer:">'+esc(L.answer)+'</a>'
                +'<a class="btn btn-ghost" href="cmd:skip">'+esc(L.skip)+'</a>'
                +'<span class="spacer"></span>'
                +'<a class="btn btn-danger" href="cmd:abort">'+esc(L.abort)+'</a>'
                +'</div>';
        }else{
            html+='<div class="opts-hint">'+esc(L.noOptions)+'</div>';
            html+='<div class="actions"><a class="btn btn-danger" href="cmd:abort">'+esc(L.abort)+'</a></div>';
        }
    }
    html+='</div>';
    root.innerHTML=html;
    refresh();
}
render();
</script>
