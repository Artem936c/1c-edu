function readData(){
  var el = document.getElementById('dashboard-data');
  var raw = el ? String(el.textContent || '').trim() : '';
  try { return (raw.indexOf('{{') === 0 || raw === '') ? MOCK_DATA : JSON.parse(raw); }
  catch(e){ return MOCK_DATA; }
}
function esc(s){ return String(s == null ? '' : s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;'); }
function kindClass(v){
  var k = String(v || '');
  if (k.indexOf('Справочник') >= 0) return 'spravochnik';
  if (k.indexOf('Метод') >= 0) return 'metod';
  if (k.indexOf('Норм') >= 0) return 'norm';
  if (k.indexOf('Видео') >= 0) return 'video';
  return '';
}
function kindLabel(v){
  var m = {'МетодическоеПособие':'Метод. пособие','НормативныйДокумент':'Норм. документ','Видеоматериал':'Видео'};
  return m[v] || v;
}
function bookCard(b){
  var sub = [];
  if (b.Автор) sub.push(esc(b.Автор));
  if (b.Год) sub.push(esc(b.Год));
  if (b.Издательство) sub.push(esc(b.Издательство));
  var tags = '';
  if (b.КлючевыеСлова){
    var kw = String(b.КлючевыеСлова).split(/[;,]/);
    for (var i=0;i<kw.length && i<5;i++){ var t = kw[i].replace(/^\s+|\s+$/g,''); if (t) tags += '<span class="tag">'+esc(t)+'</span>'; }
  }
  var meta = [];
  if (b.ISBN) meta.push('ISBN '+esc(b.ISBN));
  return '<a class="book" href="#" data-act="материал" data-id="'+esc(b.МатериалИД||'')+'" data-kind="'+esc(b.Вид)+'" data-search="'+esc(((b.Название||'')+' '+(b.Автор||'')+' '+(b.КлючевыеСлова||'')+' '+(b.Тема||'')).toLowerCase())+'">'
    + '<div class="book-cover"><svg viewBox="0 0 24 24" width="26" height="26" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M12 6.5C10.3 5 7.7 4.5 4 4.5V18c3.7 0 6.3.5 8 2 1.7-1.5 4.3-2 8-2V4.5c-3.7 0-6.3.5-8 2Z"/><path d="M12 6.5V20"/></svg></div>'
    + '<div class="book-body">'
    + '<div class="book-name">'+esc(b.Название)+'</div>'
    + '<div class="book-author">'+sub.join(' · ')+'</div>'
    + '<span class="mat-kind '+kindClass(b.Вид)+'">'+esc(kindLabel(b.Вид))+'</span>'
    + (b.Описание ? '<div class="book-desc">'+esc(b.Описание)+'</div>' : '')
    + (b.Библиография ? '<div class="book-gost">'+esc(b.Библиография)+'</div>' : '')
    + (meta.length ? '<div class="book-meta">'+meta.join(' · ')+'</div>' : '')
    + (tags ? '<div class="book-tags">'+tags+'</div>' : '')
    + '</div></a>';
}
function render(data){
  document.getElementById('lib-title').textContent = data.Заголовок || 'Встроенная библиотека';
  document.getElementById('lib-sub').textContent = 'Всего изданий: ' + (data.ВсегоМатериалов || 0) + ' · сгруппировано по темам платформы 1С:Предприятие 8.3';
  var groups = data.Группы || [];
  var body = document.getElementById('lib-body');
  if (!groups.length){ body.innerHTML = '<div class="empty-state">В библиотеке пока нет материалов.</div>'; return; }
  var h = [];
  for (var g=0; g<groups.length; g++){
    var grp = groups[g];
    var mats = grp.Материалы || [];
    h.push('<div class="theme-block"><div class="theme-title">'+esc(grp.Тема)+' · '+mats.length+'</div><div class="books">');
    for (var m=0; m<mats.length; m++) h.push(bookCard(mats[m]));
    h.push('</div></div>');
  }
  body.innerHTML = h.join('');
}
function applyFilters(){
  var qEl = document.getElementById('q');
  var q = qEl ? String(qEl.value || '').toLowerCase().replace(/^\s+|\s+$/g,'') : '';
  var activeChip = document.querySelector('.chip.active');
  var kind = activeChip ? (activeChip.getAttribute('data-kind') || '') : '';
  var books = document.querySelectorAll('.book');
  for (var i=0;i<books.length;i++){
    var b = books[i];
    var okKind = !kind || b.getAttribute('data-kind') === kind;
    var okSearch = !q || String(b.getAttribute('data-search') || '').indexOf(q) >= 0;
    b.style.display = (okKind && okSearch) ? '' : 'none';
  }
  // Скрыть пустые блоки тем.
  var blocks = document.querySelectorAll('.theme-block');
  for (var j=0;j<blocks.length;j++){
    var vis = blocks[j].querySelectorAll('.book');
    var any = false;
    for (var k=0;k<vis.length;k++){ if (vis[k].style.display !== 'none'){ any = true; break; } }
    blocks[j].style.display = any ? '' : 'none';
  }
}
function wireUI(){
  var qEl = document.getElementById('q');
  if (qEl) qEl.addEventListener('input', applyFilters);
  var chips = document.querySelectorAll('.chip');
  for (var i=0;i<chips.length;i++){
    chips[i].addEventListener('click', function(ev){
      var all = document.querySelectorAll('.chip');
      for (var j=0;j<all.length;j++) all[j].className = 'chip';
      ev.currentTarget.className = 'chip active';
      applyFilters();
    });
  }
  // Клик по карточке издания - якорь data-act="материал". Гасим переход по href="#";
  // открытие материала выполняет форма (обработчик ПолеHTMLПриНажатии по этому якорю).
  document.addEventListener('click', function(ev){
    var a = ev.target;
    while (a && a.nodeType===1 && !(a.getAttribute && a.getAttribute('data-act'))) a = a.parentNode;
    if (a && a.nodeType===1 && a.getAttribute && a.getAttribute('data-act') === 'материал') ev.preventDefault();
  }, false);
}

  // Интерактивность (поиск/фильтры) - необязательна: ошибки не должны ломать показ списка.
  try { wireUI(); } catch(e2){}
})();
